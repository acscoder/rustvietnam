Sử dụng vector

Trong các Collections thì vector là phổ biến nhất, vector là 1 mảng có thể thay đổi kích thước(size).
Ok, mở play ground https://play.rust-lang.org/ hoặc tạo file mới và compile cũng được.

fn main() {
    let cars = vec!["Toyota", "Tesla", "Lambogini"];
    println!("cars: {:?}", cars);
    let mut fruits = Vec::new();
    fruits.push("apple");
    fruits.push("tomato");
    fruits.push("pear");
    println!("fruits: {:?}", fruits);
    let last = fruits.pop();
    if let Some(last) = last {
        println!("Removed {} from {:?}", last, fruits);
    }
}

Có 2 cách khai báo vector, 1 khai báo với số phần tử có sẵn let cars = vec!["Toyota", "Tesla", "Lambogini"]; , 2 khai báo với 1 vector rỗng và thay đổi sau let mut fruits = Vec::new();
Các bạn lưu ý đoạn này

    println!("Removed {} from {:?}", last, fruits);

Trong JS có hàm splice để loại bỏ/thay thế các phần tử trong mảng rất tiện, chúng ta cũng có nó trong Rust

fn main(){
    let mut stuff = vec!["1", "2", "3", "4", "5", "6"];
    println!("Original stuff: {:?}", stuff);
    let stuff_to_insert = vec!["a", "b", "c"];
    let removed_stuff: Vec<_> = stuff.splice(1..4, stuff_to_insert).collect();
    println!("Spliced stuff: {:?}", stuff);
    println!("Removed stuff: {:?}", removed_stuff);
}

Thay thế các phần tử ở stuff bằng các phần tử stuff_to_insert , vị trí thay thế 1..4, nếu ta thay vec let stuff_to_insert = Vec::new(); là vector rỗng thì sẽ là remove các phần tử 1,2,3.

Với Vector

// ta có 1 iterator bằng hàm **iter**

Với chuỗi ta ko có hàm iter mà dùng hàm char

let mut alphabet = "ABCDEFGHIJKLMNOPQRSTUVWXYZ".chars();

// Hàm **enumerate** tạo iterator chi tiết với kí tự và index của kí tự, chúng ta thử vòng lặp for sẽ thấy

Sử dụng VecDeque

VecDeque là ngăn xếp dạng First In, First Out (FIFO) , vào trước thì lấy ra trước.Dùng playground hoặc tạo file src/bin/vecdeque.rs , thêm code dưới rồi chạy với lệnh cargo run --bin vecdeque.

   use std::collections::VecDeque;

   fn main() {
     // VecDeque là dạng ngăn xếp First-In-First-Out (FIFO) 
     // Thường xuyên, bạn sẽ sữ dụng hàm push_back để đưa vào ngăn xếp
     //  và lấy ra với hàm pop_front
    let mut orders = VecDeque::new();
    println!("A guest ordered oysters!");
    orders.push_back("oysters");

    println!("A guest ordered fish and chips!");
    orders.push_back("fish and chips");

    let prepared = orders.pop_front();
    if let Some(prepared) = prepared {
      println!("{} are ready", prepared);
    }

    println!("A guest ordered mozarella sticks!");
    orders.push_back("mozarella sticks");

    let prepared = orders.pop_front();
    if let Some(prepared) = prepared {
      println!("{} are ready", prepared);
    }

    println!("A guest ordered onion rings!");
    orders.push_back("onion rings");

    let prepared = orders.pop_front();
    if let Some(prepared) = prepared {
      println!("{} are ready", prepared);
    }

    let prepared = orders.pop_front();
    if let Some(prepared) = prepared {
      println!("{} are ready", prepared);
    }

    // Bạn có thể thay đổi thứ tự đưa vào từ push_back thành push_front (đưa phần tử vào phía trước)

    let mut sentence = VecDeque::new();
    sentence.push_back("a");
    sentence.push_front("had");
    sentence.push_back("little");
    sentence.push_front("Mary");
    sentence.push_back("Lamb");
    println!("sentence: {:?}", sentence);

    // Tương tự với lấy ra,có thể pop_back thay vì pop_front
    sentence.pop_front();
    sentence.push_front("Jimmy");
    sentence.pop_back();
    sentence.push_back("Cat");
    println!("sentence: {:?}", sentence);

    // Còn lại thì các hàm của VecDeque khá giống với vector    
    let mut some_queue = VecDeque::with_capacity(5);
    some_queue.push_back("A");
    some_queue.push_back("B");
    some_queue.push_back("C");
    some_queue.push_back("D");
    some_queue.push_back("E");
    println!("some_queue: {:?}", some_queue);

    // cái này giống với swap_remove của vector
    some_queue.swap_remove_back(2);
    println!("some_quere after swap_remove_back: {:?}",
        some_queue);

    //Tương tự nhưng là front với swap_remove_front

    some_queue.swap_remove_front(2);
    println!("some_quere after swap_remove_front: {:?}",
        some_queue);
    }

Sử dụng HashMap

HashMap là 1 mảng dạng key => value
Thử đoạn code sau là bạn sẽ thấy

use std::collections::HashMap;

fn main() {
    // đầu tiên là "key"
    // sau là "value"
    let mut tv_ratings = HashMap::new();
    // Chúng ta mapping &str với i32
    tv_ratings.insert("The IT Crowd", 8);
    tv_ratings.insert("13 Reasons Why", 7);
    tv_ratings.insert("House of Cards", 9);
    tv_ratings.insert("Stranger Things", 8);
    tv_ratings.insert("Breaking Bad", 10);

    // Kiểm tra key có tồn tại với contains_key
    let contains_tv_show = tv_ratings.contains_key("House of Cards");
    println!("Did we rate House of Cards? {}", contains_tv_show);
    let contains_tv_show = tv_ratings.contains_key("House");
    println!("Did we rate House? {}", contains_tv_show);

    // lấy giá trị value với key
    if let Some(rating) = tv_ratings.get("Breaking Bad") {
        println!("I rate Breaking Bad {} out of 10", rating);
    }

    // Nếu ta insert value 2 lần ,thì dữ liệu sẽ ghi đè
    let old_rating = tv_ratings.insert("13 Reasons Why", 9);
    if let Some(old_rating) = old_rating {
        println!("13 Reasons Why's old rating was {} out of 10", old_rating);
    }
    if let Some(rating) = tv_ratings.get("13 Reasons Why") {
        println!("But I changed my mind, it's now {} out of 10", rating);
    }

    // Xóa key/value
    let removed_value = tv_ratings.remove("The IT Crowd");
    if let Some(removed_value) = removed_value {
        println!("The removed series had a rating of {}", removed_value);
    }

    // Iterating accesses all keys and values
    println!("All ratings:");
    for (key, value) in &tv_ratings {
        println!("{}\t: {}", key, value);
    }

    // Thay đổi giá trị biến khi duyệt các phần tử
    println!("All ratings with 100 as a maximum:");
    for (key, value) in &mut tv_ratings {
        *value *= 10;
        println!("{}\t: {}", key, value);
    }

    // duyệt qua các phần tử của mảng mà không dùng tham chiếu
    for _ in tv_ratings {}
    // tv_ratings đã được sử dụng , không thể dùng tiếp được

    // Giới hạn số phần tử được dùng trong HashMap

    let mut age = HashMap::with_capacity(10);
    age.insert("Dory", 8);
    age.insert("Nemo", 3);
    age.insert("Merlin", 10);
    age.insert("Bruce", 9);

    // Duyệt qua các key
    println!("All names:");
    for name in age.keys() {
        println!("{}", name);
    }

    // Duyệt qua các values
    println!("All ages:");
    for age in age.values() {
        println!("{}", age);
    }

    // Duyệt qua các key , value và thay đổi nó
    println!("All ages in 10 years");
    for age in age.values_mut() {
        *age += 10;
        println!("{}", age);
    }

    // Đặt giá trị mặc định nếu phần tử không tồn tại
    {
        let age_of_coral = age.entry("coral").or_insert(11);
        println!("age_of_coral: {}", age_of_coral);
    }
    let age_of_coral = age.entry("coral").or_insert(15);
    println!("age_of_coral: {}", age_of_coral);
}

HashSet là một HashMap có value rỗng , HashSet = HashMap<K, ()>

use std::collections::HashSet;

fn main() {
    // Tất cả các hàm dùng trong HashSet
    // đều giống như HashMap, chỉ không có 
    // các hàm xử lí value
    let mut books = HashSet::new();
    books.insert("Harry Potter and the Philosopher's Stone");
    books.insert("The Name of the Wind");
    books.insert("A Game of Thrones");

    // A HashSet sẽ không duplicate phần tử
    // nhưng sẽ trả về true/false nếu phần tử thêm vào là mới hoặc không
    let is_new = books.insert("The Lies of Locke Lamora");
    if is_new {
        println!("We've just added a new book!");
    }

    let is_new = books.insert("A Game of Thrones");
    if !is_new {
        println!("Sorry, we already had that book in store");
    }

    // kiểm tra contains key
    if !books.contains("The Doors of Stone") {
        println!("We sadly don't have that book yet");
    }

    // Xóa phần tử
    let was_removed = books.remove("The Darkness that comes before");
    if !was_removed {
        println!("Couldn't remove book; We didn't have it to begin with");
    }
    let was_removed = books.remove("Harry Potter and the Philosopher's Stone");
    if was_removed {
        println!("Oops, we lost a book");
    }

    // So sánh 2 HashSets

    let one_to_five: HashSet<_> = (1..6).collect();
    let five_to_ten: HashSet<_> = (5..11).collect();
    let one_to_ten: HashSet<_> = (1..11).collect();
    let three_to_eight: HashSet<_> = (3..9).collect();

    // Kiểm tra nếu 2 HashSets không có phần tử nào trùng
    let is_disjoint = one_to_five.is_disjoint(&five_to_ten);
    println!(
        "is {:?} disjoint from {:?}?: {}",
        one_to_five,
        five_to_ten,
        is_disjoint
    );
    let is_disjoint = one_to_five.is_disjoint(&three_to_eight);
    println!(
        "is {:?} disjoint from {:?}?: {}",
        one_to_five,
        three_to_eight,
        is_disjoint
    );

    // Kiểm tra nếu 1 HashSet là tập hợp con của 1 HashSet khác
    let is_subset = one_to_five.is_subset(&five_to_ten);
    println!(
        "is {:?} a subset of {:?}?: {}",
        one_to_five,
        five_to_ten,
        is_subset
    );
    let is_subset = one_to_five.is_subset(&one_to_ten);
    println!(
        "is {:?} a subset of {:?}?: {}",
        one_to_five,
        one_to_ten,
        is_subset
    );

    // Kiểm tra nếu 1 HashSet là tập hợp cha của 1 HashSet khác.
    let is_superset = three_to_eight.is_superset(&five_to_ten);
    println!(
        "is {:?} a superset of {:?}?: {}",
        three_to_eight,
        five_to_ten,
        is_superset
    );
    let is_superset = one_to_ten.is_superset(&five_to_ten);
    println!(
        "is {:?} a superset of {:?}?: {}",
        one_to_ten,
        five_to_ten,
        is_superset
    );

    // 1 tập hơp các phần tử khác nhau của 2 HashSets 

    let difference = one_to_five.difference(&three_to_eight);
    println!(
        "The difference between {:?} and {:?} is {:?}",
        one_to_five,
        three_to_eight,
        difference
    );

    //1 tập hơp các phần tử không chung nhau của 2 HashSets
    let symmetric_difference = one_to_five.symmetric_difference(&three_to_eight);
    println!(
        "The symmetric difference between {:?} and {:?} is {:?}",
        one_to_five,
        three_to_eight,
        symmetric_difference
    );

    // 1 tập hơp các phần tử chung của 2 HashSets 
    let intersection = one_to_five.intersection(&three_to_eight);
    println!(
        "The intersection difference between {:?} and {:?} is {:?}",
        one_to_five,
        three_to_eight,
        intersection
    );

    // Lấy tất cả các phần tử từ 2 hashsets
    let union = one_to_five.union(&three_to_eight);
    println!(
        "The union difference between {:?} and {:?} is {:?}",
        one_to_five,
        three_to_eight,
        union
    );
}
